INITIALIZATION
--------------
from services.video_agent.pro import AssetLibrary, AssetType, AssetCategory

library = AssetLibrary(storage_dir="/path/to/assets")

IMPORT SINGLE ASSET
-------------------
asset = library.import_asset(
    source_path="video.mp4",
    asset_type=AssetType.VIDEO,
    name="My Video",
    category=AssetCategory.PRODUCT_SHOT,
    tags={"product", "4k", "demo"},
    description="Product demo video",
    generate_thumbnail=True
)

BULK IMPORT
-----------
videos = library.bulk_import(
    folder_path="/videos",
    asset_type=AssetType.VIDEO,
    category=AssetCategory.STOCK_FOOTAGE,
    recursive=True,
    file_extensions=['.mp4', '.mov', '.avi']
)

IMPORT FROM URL
---------------
asset = library.import_from_url(
    url="https://example.com/video.mp4",
    asset_type=AssetType.VIDEO,
    tags={"downloaded", "stock"}
)

SEARCH ASSETS
-------------
# Simple text search
results = library.search_assets(query="product")

# Search by type
videos = library.search_assets(asset_type=AssetType.VIDEO)

# Search by category
music = library.search_assets(category=AssetCategory.MUSIC)

# Search by tags
tagged = library.search_assets(tags={"4k", "product"})

# Complex search
results = library.search_assets(
    query="ocean",
    asset_type=AssetType.VIDEO,
    category=AssetCategory.B_ROLL,
    min_resolution=(1920, 1080),
    max_duration=60.0,
    has_audio=False,
    sort_by="created_at",
    reverse=True
)

CLOUD STORAGE
-------------
from services.video_agent.pro import CloudStorageManager

# Setup
cloud = CloudStorageManager(
    bucket_name="my-bucket",
    credentials_path="/path/to/gcs-credentials.json"
)

library = AssetLibrary(
    storage_dir="/local/assets",
    cloud_storage=cloud
)

# Upload
library.upload_to_cloud(asset.id, include_thumbnail=True)

# Get URL
url = cloud.get_public_url(asset.cloud_path)

METADATA
--------
# Get metadata
metadata = library.get_asset_metadata(asset.id)

# Video metadata
if isinstance(metadata, VideoMetadata):
    print(f"Resolution: {metadata.resolution}")
    print(f"Duration: {metadata.duration}s")
    print(f"FPS: {metadata.fps}")
    print(f"Codec: {metadata.codec.value}")
    print(f"4K: {metadata.is_4k}")
    print(f"HD: {metadata.is_hd}")

# Audio metadata
if isinstance(metadata, AudioMetadata):
    print(f"Duration: {metadata.duration}s")
    print(f"Sample Rate: {metadata.sample_rate} Hz")
    print(f"Channels: {metadata.channels}")
    print(f"Stereo: {metadata.is_stereo}")

PRO TIPS
--------
1. Always generate thumbnails for quick preview
2. Use consistent tagging conventions
3. Mark frequently used assets as favorites
4. Upload important assets to cloud for backup
5. Use specific categories rather than generic tags
6. Version important assets rather than overwriting
7. Use bulk import for initial library setup
8. Track usage to identify popular assets
